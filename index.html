import duckdb
import pandas as pd

df = pd.DataFrame({
'a': [1,2,3,4],
'b': [True, None, False,True],
'c': [42, 21, 13, 14]
})

const = duckdb.ConstantExpression('hello')
res = duckdb.df(df).select(const).fetchall()
print(res)
